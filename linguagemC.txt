float RaizQuadrada(float N)
{
    int i;
    float RQ=1.0;
    for(i=0;i<100;i++)
    {
        RQ=0.5*(RQ+N/RQ);
    }
    return RQ;
}
 
 
int main()
{
    float RQ;
    float N;
    printf("Escreva o número: ");
    scanf("%f", &N);
    RQ = RaizQuadrada(N);
    printf("\n A raiz quadrada de %f é %f \n", N, RQ);
}

int palindromo(int v[], int tam)
{
	int i;
	int stop=0;

	for(i=0;i<tam/2;i++)
	{
		if(v[i] != v[tam-1-i])
		{
			stop=1;
			break;
		}
	}

	if(stop==0)
	{
		printf("O vetor é um palíndromo \n");	
	}
	else
	{
		printf("O vetor não é um palíndromo \n");
	}
}

#include <stdio.h>

unsigned long long superfatorial(unsigned long long N)
{
	unsigned long long fat=1,SF=1;
	int i,j; 

	if(N==0)
	{
		return 1;
	}
	
	for(j=2;j<=N;j++)
	{
		fat=1;
		for(i=2;i<=j;i++)
		{
			fat=fat*i;
		}
		SF=SF*fat;
	}

	return SF;
}

int main()
{
	unsigned long long n;
	unsigned long long SF;
	printf("Escreva o número: ");
	scanf("%llu",&n);
	SF=superfatorial(n);
	printf("O super fatorial de %llu é %llu \n",n,SF);
}

{
	int senha=0, digitos[6], i, pot=1000000;
	digitos[0]=(nome[0]%dia)%10;
	digitos[1]=(nome[1]%dia)%10;
